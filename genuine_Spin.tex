\section{Coherent continuation representation of the Spin groups}

Let $G_{\bC} = \Spin(m,\bC)$ be the complex spin group. Let $Q$ be the integral
weight lattic of $G_{\bC}$, and $X$ be the weight lattic of $\SO(m,\bC)$ and it
is identified with the sub-lattic of $Q$ with index $2$.

Let $\Lambda \in \fhh^{*}/Q$. Then $\Lambda = \Lambda_{1} \cup \Lambda_{2}$
where $\Lambda_{1},\lambda_{2} \in \fhh^{*}/X$.

For any real form of $G_{\bC}$, i.e. a subgroup of $G_{\bC}$ such that $\Lie(G)$
is a real from of $\Lie(G_{\bC})$. Let $\cK_{\lambda}(G)$ be the set
Grothendieck of finite-lenght $G$-modules having infinitesimial character
$\lambda\in \fhh^{*}$.

Let $\cKgen(G)$ to be the Grothendieck group of genuine representations of $G$
(i.e. generated by irreducible representations of $G$ not factor through the
corresponding orthogonal group) Let $G'$ be the image of $G$ under the natural
quotient $\Spin(m,\bC)\rightarrow \SO(m,\bC)$.

Let $\cKcls(G)$ be the Grothendieck groups of representations of $G$ factor
through $G'$.

Let
\[
  \Coh_{\Lambda_{1},\Lambda_{2}}(G):= \Set{ \Psi\colon \Lambda \rightarrow \cK(G)|
    \begin{array}{l}
      \forall \lambda \in \Lambda, \Psi(\lambda)\in \cK_{\lambda}(G),\\
      \Im(\Psi|_{\Lambda_{1}})\subset \cKcls(G),\\
      \Im(\Psi|_{\Lambda_{2}})\subset \cKgen(G).
    \end{array}
  }
\]



The following lemma is clear
\begin{lem}\label{lem:iso}
  Restrictions induces the following isomorphisms of $W(\Lambda)$-module,
  \[
    \Coh_{\Lambda_{1}}(G') \xleftarrow{ \ \ \ \cong \ \ } \Coh_{\Lambda_{1},\Lambda_{2}}(G) \xrightarrow{ \ \ \cong \ \ \ } \Cohgen_{\Lambda_{2}}(G).
  \]
\end{lem}
\begin{proof}
  The $W(\Lambda)$-equivariance is clear. The isomorphism follows from that
  evaluation at any regular infinitesimial character is gives bijection.
\end{proof}


Let $\ckG$ be the Langlands dual of $G$. Without of loss of generality, we
assume $\ckcO = 2\ckcO'_{b}\sqcup_{\mathrm{row}}\ckcO_{g}$ is a nilpotent orbit
of $G$.

As a corollary of \Cref{lem:iso}, we have
\[

\subsection{Quaternianic Spin group}

Suppose $G = \Spin^{*}(2n)$. We the notion of $n_{g}:= \abs{\ckcO_{g}}/2$ and
$n_{\ckcO'_{b}}$.

We have
\[
  \Cohgen_{\lamck+X}(G) \cong \Ind_{W'_{n_{g}}\times W'_{n_b}}^{W_{\Lambda}}
  \cCb_{n_{g}}\times \cCg_{n_{b}}
\]


\begin{lem}
  If $n_{g}\neq 0$, then $[\tau_{\wp}:\cCb_{n_{g}}]_{W'_{n_{g}}} = 0$ for every $\wp\in \PP(\ckcO)$.
\end{lem}
\begin{proof}
  Just look at the shape of $\tau_{\wp}$, we have
  $\bfcc_{1}(\tau_{\wp,L}) = r_{1}+1 > r_{i}-1 = \bfcc_{1}(\tau_{\wp,R})$ ($i=2$
  if $(2,3)\notin \wp$ and $i=3$ otherwise).
  On the other hand, every representations accure in $\cCb_{n_{g}}$ has
  identical left and right diagrams.
\end{proof}

Let $\cO = [2r'_{1},2r'_{1}, 2r'_{2},2r'_{2},\cdots,2r_{l},2r_{l}]_{col}$.
Note that $\cO$ has codimension $2$-boundary.

\begin{lem}
  Assuming the label $I/II$ is matched for $\ckcO$ and the group $G$.
  If $n_{g}=0$, then $[\tau_{b}:\cCg_{n_{b}}]_{W'_{n_{b}}} = \#\Nil_{\cO}(G)$.
\end{lem}
\begin{proof}
  Note that the shape of left and right diagrams are identical.
  It is clear that all painting have the form that a the left diagram some ``s''
  are painted in the end of each column, for $i$-th column at most
  $(r'_{i}-r'_{i+1})$ times. In the right diagram, paint $r$ if and only $s$
  occurs in the corresponding place in the left diagram.
  All other places, one paint dots.

  Now the number of possible painted bipartitions are
  \[
    \prod_{i}(r'_{i}-r'_{i+1}+1).
  \]
  It is the same as the number of valid signed Young diagrams has shape $\cO$.
  This finished the proof.
\end{proof}

\subsection{Even orthogonal groups}
Let $G = \wtSpin(p,q)$ wich is the spin cover of $\SO(p,q)$.

We have
\[
  \Cohgen_{\lamck+X} (G) \cong \cCb_{n_{g}}\otimes \cCg_{p-n_{g},q-n_{g}}.
\]




\begin{lem}
  $[\tau_{b}:\cCg_{p,q}] =  1$ if $p=q$ and $=0$ otherwise.
\end{lem}
\begin{proof}
  Look at the painted bipartition rules, it is clear that the only possible
  filling is put dots everywhere else. In this case $p=q$.
\end{proof}



\begin{lem}
  $[\tau_{\wp}:\cCb_{n_{g}}] =  1$ only if $r_{2i-1} = r_{2i}$.
\end{lem}
\begin{proof}

  Note that we always have
  $r_{2i+1}+1$ or
  $r_{2i}+1 = \bfcc_{i}(\tau_{\wp,L}) > \bfcc_{i}(\tau_{\wp,R}) = r_{2i+2}-1$
  or $r_{2i+3}-1$.

  By painting rules, the only possible filling is to put  ``c'' at the bottom
  of each columns in the left diagram and  dots on other places.
  In this case, $r_{2i+1}=r_{2i+2}$, $\wp=\emptyset$ and the multiplicity is $1$.
\end{proof}

\begin{cor}
  There is a unique special unipotent representations of $\wtSpin(p,q)$ for each very even
  nilpotent orbit $\ckcO$ if $p=q$.
\end{cor}

\begin{lem}
  $[\tau_{\wp}:\cCb_{n_{g}}] =  1$ only if $r_{2i-1} = r_{2i}$.
\end{lem}
\begin{proof}
  Note that we always have
  $r_{2i+1} $ or $r_{2i} = \bfcc_{i}(\tau_{\wp,L}) \leq \bfcc_{i}(\tau_{\wp,R})$
  since $r_{2i+1}\leq r_{2i}\leq r_{2i-1}$.

  By painting rules, the attachments of ``c'' or ``d'' are on the left diagrams.

  The only possible filling is to put a dot on every boxes.
  In this case, $r_{2i-1}=r_{2i}$, $\wp=\emptyset$ and the multiplicity is $1$.
\end{proof}
